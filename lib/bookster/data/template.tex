% Part covers
\newcommand{\makepartcover}[4]{
  \cleardoublepage
  \stepcounter{part}
  \phantomsection 
  \addcontentsline{toc}{part}{\protect\numberline{\thepart}#1}
  
  \thispagestyle{empty}
  
  {
    % Epigraph Styling Configuration
    \setlength{\epigraphwidth}{0.7\textwidth}
    \renewcommand{\epigraphrule}{0pt}
    
    % Left-align the text and the source
    \renewcommand{\textflush}{flushleft}
    \renewcommand{\sourceflush}{flushleft}
    
    % Custom styling for the source (dash and small caps or italics)
    \renewcommand{\epigraphsize}{\small}
    
    \vspace*{\fill}
    
    \begin{center}
      % Part Number
      {\large\textsc{\partname}\ \thepart} 
      \vspace{1cm} 
      
      % Part Title
      \begin{minipage}{0.9\textwidth}
        \centering
        \Huge\itshape #1\par
        \vspace{2cm}
        \ifblank{#4}{}{
          \pgfornament[width=3cm, opacity=0.1]{#4}
        }
      \end{minipage}
    \end{center}
    
    \vspace{2cm} 
    \renewcommand{\sourceflush}{flushright}

    % Cover quote and quote author
    \hspace{0.15\textwidth}
    \begin{minipage}{0.8\textwidth}
      \if\relax\detokenize{#2}\relax\else
        \makeatletter
        \def\@epirule{\rule[.5ex]{1pt}{60pt}}
        \makeatother
        \epigraph{\textit{#2}}{
          \hspace{1em} \fontsize{9}{11}\selectfont \textit{--- #3}\hspace{2pt}
        }
      \fi
    \end{minipage}

    \vspace*{\fill}
  }
  \newpage
}
